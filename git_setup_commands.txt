# STEP 1: Install Git CLI (Command Line Interface)
# NOTE: GitHub Desktop is different from Git CLI!
# 1. Go to https://git-scm.com/downloads (correct link)
# 2. Click "Download for Windows" 
# 3. Download "Git for Windows" (the CLI version)
# 4. Use default settings (just click "Next" through everything)
# 5. RESTART VS Code completely after installation
# 6. Test with: git --version

# ALTERNATIVE: Use GitHub CLI (since you have 'gh' installed)
# This creates the repo on GitHub and sets up Git locally in one go:

gh repo create sm --public --source=. --description="SM1.5 - Scientific Measurement Framework"

# STEP 2: Commands to run after installing Git
# (Run these one by one in PowerShell from the SM1.5 directory)

# Initialize git repository
git init

# Add your GitHub username and email (replace with your info)
git config user.name "Your Name"
git config user.email "your.email@example.com"

# Add all files (respecting .gitignore)
git add .

# Make initial commit
git commit -m "Initial commit: SM1.5 scientific measurement framework"

# Add the GitHub remote (replace YOUR_USERNAME with your GitHub username)
git remote add origin https://github.com/YOUR_USERNAME/sm.git

# Push to GitHub (beta branch for development)
git branch -M beta
git push -u origin beta

# Later, when ready for release, you can merge beta to main:
# git checkout main (or: git switch main)
# git merge beta
# git push origin main
